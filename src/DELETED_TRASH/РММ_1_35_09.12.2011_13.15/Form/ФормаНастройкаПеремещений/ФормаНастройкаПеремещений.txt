
&НаКлиенте
Процедура СохранитьНастройки(Команда)
	
	Если ПроверитьЗаполненияФормы() Тогда
		Возврат;
	КонецЕсли;
	
	СтруктураПередачи = Новый Структура;
	СтруктураПередачи.Вставить("СкладОтправитель", СкладОтправитель);
	СтруктураПередачи.Вставить("СкладПолучатель", СкладПолучатель);
	
	Закрыть(СтруктураПередачи);
	
КонецПроцедуры

&НаКлиенте
Функция ПроверитьЗаполненияФормы()
	
	Если СкладПолучатель.Пустая() Тогда
		
		Сообщение = Новый СообщениеПользователю;
		Сообщение.Текст = "Поле необходимо заполнить.";
		Сообщение.ПутьКДанным = "СкладПолучатель";
		
		Сообщение.Сообщить();
		
		Возврат Истина;
		
	КонецЕсли;
	
	Возврат Ложь;
	
КонецФункции

&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	СкладОтправитель 	= Параметры.СкладОтправитель;
	СкладПолучатель		= Параметры.СкладПолучатель;
	
КонецПроцедуры
