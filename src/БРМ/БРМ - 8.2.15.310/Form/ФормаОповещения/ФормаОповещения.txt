&НаКлиенте
Функция ПолучитьСписокТестов(ЮнитТестирование) Экспорт
	
	ЮнитТест = ЮнитТестирование;
	
	СписокТестов = Новый Массив;	
	
	ПараметрыТеста = Новый Структура("ИмяТеста, Транзакция, Параметр", "Тест_Заглушка");
	СписокТестов.Добавить(ПараметрыТеста);
	
	Возврат СписокТестов;
	
КонецФункции

&НаКлиенте
Функция Тест_Заглушка(Параметр) Экспорт
	ЮнитТест.ПроверитьИстину(Истина,);
КонецФункции // Тест_Заглушка()

&НаКлиенте
Процедура ПриОткрытии(Отказ)
	ТекущаяДата = ТекущаяДата();
	Элементы.Закрыть.Заголовок = "Все зрозуміло (2)";
	ПодключитьОбработчикОжидания("ОбновитьВидимостьКнопки", 1);
КонецПроцедуры

&НаКлиенте
Процедура ОбновитьВидимостьКнопки()
	Элементы.Закрыть.Заголовок = "Все зрозуміло (" +  (2 - (ТекущаяДата() - ТекущаяДата))  + ")";
	Если ТекущаяДата() >= ТекущаяДата + 2 Тогда
		Элементы.Закрыть.Доступность = Истина;
		ОтключитьОбработчикОжидания("ОбновитьВидимостьКнопки");
	КонецЕсли;		
КонецПроцедуры


&НаКлиенте
Процедура ПередЗакрытием(Отказ, СтандартнаяОбработка)
	Если ТекущаяДата() < ТекущаяДата + 2 Тогда
		Отказ = Истина;		
	КонецЕсли;
КонецПроцедуры
