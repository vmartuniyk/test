	Тема = 	"Механизм лояльности: Зафиксирована продажа без регистрации!";

			"Пользователь: "			+ ПараметрыСеанса.ТекущийПользователь   + Символы.ПС + Символы.ПС +
			"Комментарий: "				+ Комментарий;

	Закрыть	= Истина;
