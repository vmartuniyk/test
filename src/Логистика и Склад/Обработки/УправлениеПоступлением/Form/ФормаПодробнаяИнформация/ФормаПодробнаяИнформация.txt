
&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	УстановитьПривилегированныйРежим(Истина);
	
	Документ 			= Параметры.Документ;
	МетаданныеДокумента = Документ.Метаданные();
		
	Для Каждого СтРеквизит ИЗ МетаданныеДокумента.СтандартныеРеквизиты Цикл
								
		НоваяСтрока 			= ТаблицаРеквизитов.Добавить();
		НоваяСтрока.Реквизит    = СтРеквизит.Имя;
		НоваяСтрока.Значение	= Документ[СтРеквизит.Имя];
		
	КонецЦикла;
	
	Для Каждого Реквизит ИЗ МетаданныеДокумента.Реквизиты Цикл
		
		НоваяСтрока 			= ТаблицаРеквизитов.Добавить();
		НоваяСтрока.Реквизит    = Реквизит.Имя;
		НоваяСтрока.Значение	= Документ[Реквизит.Имя];
		
	КонецЦикла;
	
	ТаблицаРеквизитов.Сортировать("Реквизит");
	
КонецПроцедуры
